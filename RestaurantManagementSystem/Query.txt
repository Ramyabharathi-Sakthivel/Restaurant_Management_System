CREATE TABLE RESTAURANT_TB (
    RECORDID      VARCHAR2(12 BYTE) PRIMARY KEY,
    CUSTOMERNAME  VARCHAR2(40 BYTE) NOT NULL,
    FOODITEM      VARCHAR2(40 BYTE) NOT NULL,
    QUANTITY      NUMBER NOT NULL,
    ORDER_DATE    DATE NOT NULL,
    TOTALAMOUNT   NUMBER,
    REMARKS       VARCHAR2(100 BYTE)
);

CREATE SEQUENCE RESTAURANT_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;
